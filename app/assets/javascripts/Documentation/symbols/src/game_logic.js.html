  6 var clear;
  7 var correct_words = "";
  8 var start_time = new Date().getTime();
  9 var wpm;
 10 var total_keystrokes = 0;
 11 var incorrect_keystrokes = 0;
 12 
 13 const line_width = 60;
 14 
 15 /*
 16  * This Function Implements the GameLogic
 17  */
 18 function GameLogic()
 19 {
 20   this.game_text = "";
 21   this.current_line = "Looking for games to join";
 22   this.game_status = false;
 23   this.pos = 0;
 24 
 25   this.init = function(text)
 26   {
 27     this.game_text = text;
 28     this.get_line();
 29   };
 30 
 31   this.process_input = function(input)
 32   {
 33     var text_arr  = this.current_line.split("");
 34     var input_arr = input.split("");
 35 
 36     var r   = [];
 37     var b = correct_words;
 38     var g;
 39 
 40     for( i = 0; i < input_arr.length; i++)
 41     {
 42       if( text_arr[i+correct_words.length] == input_arr[i] && r.length == 0)
 43       {
 44         b += input_arr[i];
 45         if( input_arr[i] == ' ' || (correct_words.length + input.length) == this.current_line.length )
 46         {
 47           total_keystrokes += input_arr.length;
 48           clear = true;
 49           correct_words += input;
 50           input = "";
 51           wpm = calculate_wpm();
 52         }
 53       }
 54       else
 55       {
 56         total_keystrokes += input_arr.length;
 57         incorrect_keystrokes += input_arr.length;
 58         r.push(text_arr[i+correct_words.length]);
 59       }
 60     }
 61 
 62     console.log("Line len: " + this.current_line.trim() + " correct: " + correct_words.trim());
 63     if ( this.current_line.trim().length == correct_words.trim().length )
 64     {
 65       this.pos += correct_words.length;
 66       accuracy();
 67       this.get_line();
 68       text_arr = this.current_line.split("");
 69       correct_words = '';
 70       clear = true;
 71       r = [];
 72       b = [];
 73     }
 74 
 75     g = text_arr.slice(correct_words.length+input.length).join("");
 76 
 77     return {advance: clear, black:b, red:r.join(""), gray:g, p:this.pos};
 78   };
 79 
 80   this.get_line = function()
 81   {
 82     if(this.pos+line_width < this.game_text.length)
 83     {
 84       this.current_line = this.game_text.slice(this.pos, this.pos+line_width);
 85       console.log("Updated line: " + this.current_line);
 86     }
 87   };
 88 
 89   this.update_line = function()
 90   {
 91     console.log("Pos: " + this.pos);
 92 
 93     this.current_line = this.game_text.slice(this.pos, this.pos+line_width);
 94     console.log("Updated line: " + this.current_line);
 95   };
 96 }
 97 
 98 /**
 99  * @return  gwpm - the gross words per minute
100  */
101 function calculate_wpm()
102 {
103   var end_time = new Date().getTime();
104   var diff = end_time - start_time;
105   diff = diff/60000; // Convert to Minutes
106   total_entries = correct_words.length;
107   var gwpm = (total_entries/5)/ diff; //Gross Words Per Minute
108 
109   return gwpm;
110 
111 }
112 
113 /*
114 * @reinst
115 *
116 
117  */
118 function accuracy()
119 {
120   document.getElementById("accuracy").innerHTML = "Accuracy: "
121      + (Math.round((((total_keystrokes-incorrect_keystrokes)/total_keystrokes)*100) * 100) / 100 ) + "%";
122 }
